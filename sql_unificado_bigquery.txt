INSERT INTO `proyecto-bi-488218.datos_clima.datos_sectoriales`
(
  timestamp,
  session_id,
  sector,
  puerto,
  condiciones_mar,
  viento_velocidad,
  viento_direccion, 
  temperatura_agua,
  altura_olas,
  visibilidad,
  ai_analysis,
  metadata
)
VALUES
(
  -- ✅ Timestamp con validación y fallback
  COALESCE(
    SAFE.PARSE_TIMESTAMP('%Y-%m-%dT%H:%M:%E*SZ', @timestamp), 
    CURRENT_TIMESTAMP()
  ),
  
  -- ✅ Session ID con fallback único
  COALESCE(
    NULLIF(@session_id, ''), 
    CONCAT('naval-auto-', FORMAT_TIMESTAMP('%Y%m%d%H%M%S', CURRENT_TIMESTAMP()))
  ),
  
  -- ✅ Sector fijo
  'NAVAL',
  
  -- ✅ Puerto con validación de longitud
  CASE 
    WHEN LENGTH(@puerto) > 100 THEN SUBSTR(@puerto, 1, 100)
    WHEN @puerto IS NULL OR @puerto = '' THEN 'Puerto no especificado'
    ELSE @puerto
  END,
  
  -- ✅ Condiciones del mar normalizadas
  CASE 
    WHEN UPPER(@condiciones_mar) IN ('CALMO', 'TRANQUILO', 'CALM') THEN 'Calmo'
    WHEN UPPER(@condiciones_mar) IN ('MODERADO', 'MODERATE') THEN 'Moderado'
    WHEN UPPER(@condiciones_mar) IN ('AGITADO', 'ROUGH') THEN 'Agitado'
    WHEN UPPER(@condiciones_mar) IN ('PELIGROSO', 'DANGEROUS') THEN 'Peligroso'
    WHEN @condiciones_mar IS NULL THEN 'Sin datos'
    ELSE REGEXP_REPLACE(@condiciones_mar, r'[^a-zA-ZáéíóúÁÉÍÓÚñÑ\s]', '')
  END,
  
  -- ✅ Velocidad viento con validación de rango
  CASE 
    WHEN SAFE_CAST(@viento_velocidad AS FLOAT64) BETWEEN 0 AND 200 THEN SAFE_CAST(@viento_velocidad AS FLOAT64)
    WHEN @viento_velocidad IS NULL THEN NULL
    ELSE 0.0
  END,
  
  -- ✅ Dirección viento normalizada
  CASE 
    WHEN UPPER(@viento_direccion) IN ('N', 'NORTE', 'NORTH') THEN 'Norte'
    WHEN UPPER(@viento_direccion) IN ('S', 'SUR', 'SOUTH') THEN 'Sur'
    WHEN UPPER(@viento_direccion) IN ('E', 'ESTE', 'EAST') THEN 'Este'
    WHEN UPPER(@viento_direccion) IN ('O', 'W', 'OESTE', 'WEST') THEN 'Oeste'
    WHEN UPPER(@viento_direccion) IN ('NE', 'NORESTE', 'NORTHEAST') THEN 'Noreste'
    WHEN UPPER(@viento_direccion) IN ('NO', 'NW', 'NOROESTE', 'NORTHWEST') THEN 'Noroeste'
    WHEN UPPER(@viento_direccion) IN ('SE', 'SURESTE', 'SOUTHEAST') THEN 'Sureste'
    WHEN UPPER(@viento_direccion) IN ('SO', 'SW', 'SUROESTE', 'SOUTHWEST') THEN 'Suroeste'
    ELSE 'Variable'
  END,
  
  -- ✅ Temperatura agua con validación (-5°C a 50°C)
  CASE 
    WHEN SAFE_CAST(@temperatura_agua AS FLOAT64) BETWEEN -5 AND 50 THEN SAFE_CAST(@temperatura_agua AS FLOAT64)
    WHEN @temperatura_agua IS NULL THEN NULL
    ELSE 20.0
  END,
  
  -- ✅ Altura olas con validación (0 a 30 metros)
  CASE 
    WHEN SAFE_CAST(@altura_olas AS FLOAT64) BETWEEN 0 AND 30 THEN SAFE_CAST(@altura_olas AS FLOAT64)
    WHEN @altura_olas IS NULL THEN NULL
    ELSE 0.0
  END,
  
  -- ✅ Visibilidad con validación (0 a 50 km)
  CASE 
    WHEN SAFE_CAST(@visibilidad AS FLOAT64) BETWEEN 0 AND 50 THEN SAFE_CAST(@visibilidad AS FLOAT64)
    WHEN @visibilidad IS NULL THEN NULL
    ELSE 10.0
  END,
  
  -- ✅ AI Analysis con sanitización
  CASE 
    WHEN LENGTH(@ai_analysis) > 5000 THEN SUBSTR(@ai_analysis, 1, 5000)
    WHEN @ai_analysis IS NULL OR @ai_analysis = '' THEN 'Análisis no disponible'
    ELSE REGEXP_REPLACE(@ai_analysis, r'[^\w\sáéíóúÁÉÍÓÚñÑ.,;:()\-]', '')
  END,
  
  -- ✅ Metadata como JSON string
  COALESCE(@metadata, '{}')
);
